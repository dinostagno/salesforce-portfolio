String soql =
'SELECT Id FROM Account WHERE Industry = \'Technology\'';

EnterpriseBatchRequest request =
    new EnterpriseBatchRequest(
        soql,
        200,
        true
    );

System.schedule(
    'Enterprise Night Processing',
    cron,
    new EnterpriseBatchJobScheduler(request)
);
